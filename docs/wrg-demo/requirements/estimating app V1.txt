// --- Gemini API Utilities ---
async function searchLocalLaborRates(zipCode) {
  const systemPrompt = `Procurement Specialist for Project Aries. Find market labor rates for commercial furniture installation in zip code: ${zipCode}. Return JSON: nonUnion, union, sourceSummary.`;
  const payload = {
    contents: [{ parts: [{ text: `Labor rates zip: ${zipCode}` }] }],
    systemInstruction: { parts: [{ text: systemPrompt }] },
    tools: [{ "google_search": {} }],
    generationConfig: { responseMimeType: "application/json" }
  };
  const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(payload)
  });
  const result = await response.json();
  return JSON.parse(result.candidates?.[0]?.content?.parts?.[0]?.text || '{}');
}

async function callGeminiAI(base64Data, mimeType, availableCategories, correctionComments = "") {
  let refinementInstruction = correctionComments ? `CRITICAL CORRECTION: "${correctionComments}". Adjust extraction to address these specific points.` : "";
  const systemPrompt = `Senior Furniture Spec Specialist for Project Aries. Extract items to categories: ${Object.keys(availableCategories).join(', ')}. Extract client info and identify multi-option sets. ${refinementInstruction} Return JSON.`;
  const payload = {
    contents: [{ role: "user", parts: [{ text: "Extract spec JSON." }, { inlineData: { mimeType, data: base64Data } }] }],
    systemInstruction: { parts: [{ text: systemPrompt }] },
    generationConfig: { responseMimeType: "application/json" }
  };
  const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(payload)
  });
  const result = await response.json();
  return JSON.parse(result.candidates?.[0]?.content?.parts?.[0]?.text || '{"items":[], "clientInfo": {}}');
}

// --- Calculation Logic ---
const calculateInstallHours = (lineItems, variables, config) => {
  let baseHours = 0;
  lineItems.forEach(item => {
    const category = config.categories[item.categoryId];
    if (category) {
      const sub = item.subCategoryId ? category.subcategories?.[item.subCategoryId] : null;
      baseHours += ((sub ? sub.rate : category.rate) * (parseFloat(item.quantity) || 0));
    }
  });
  let adj = baseHours;
  if (variables.noElevator) adj *= config.multipliers.STAIR_CARRY;
  if (variables.afterHours) adj *= config.multipliers.AFTER_HOURS;
  if (variables.siteProtection) adj *= config.multipliers.SITE_PROTECTION;
  const rate = variables.isUnion ? config.rates.UNION : config.rates.NON_UNION;
  const cost = adj * rate;
  const sales = cost / (1 - (config.defaultMargin || 0.35));
  const crew = Math.ceil(adj / (8 * (parseFloat(variables.duration) || 1)));
  return { 
    baseHours: baseHours.toFixed(2), 
    totalHours: adj.toFixed(2), 
    totalCost: cost.toFixed(2), 
    salesPrice: sales.toFixed(2), 
    grossMargin: (sales - cost).toFixed(2), 
    crewSize: crew, 
    hourlyRate: rate 
  };
};

                  <button onClick={() => setLineItems([...lineItems, { id: crypto.randomUUID(), categoryId: 'SYSTEMS', subCategoryId: '', description: '', quantity: 1 }])} className="text-xs flex items-center gap-2 bg-slate-900 text-white px-6 py-3 rounded-2xl font-black uppercase tracking-wider shadow-lg hover:bg-slate-800 transition-all"><Plus size={16} /> Add Line</button>
                </div>
              </div>
              <div className="overflow-x-auto text-left">
                <table className="w-full text-left table-fixed min-w-[1200px]">
                  <thead>
                    <tr className="text-xs uppercase text-slate-400 font-black border-b border-slate-100">
                      <th className="px-10 py-8 w-[20%] tracking-widest">Group</th>
                      <th className="px-10 py-8 w-[20%] tracking-widest">Product Line</th>
                      <th className="px-10 py-8 w-[40%] tracking-widest text-left">Description Details</th>
                      <th className="px-10 py-8 w-[15%] text-center tracking-widest">QTY</th>
                      <th className="px-6 py-8 w-[5%]"></th>
                    </tr>
                  </thead>
                  <tbody className="divide-y divide-slate-50 text-left">
                    {lineItems.map((item) => (
                      <tr key={item.id} className="group hover:bg-indigo-50/40 transition-colors">
                        <td className="px-10 py-8 align-top text-left">
                          <select value={item.categoryId} onChange={(e) => updateLineItem(item.id, 'categoryId', e.target.value)} className="w-full bg-transparent border-none text-xs font-black text-slate-900 p-0 focus:ring-0 cursor-pointer">{Object.values(config.categories).map(cat => <option key={cat.id} value={cat.id}>{cat.label}</option>)}</select>
                        </td>
                        <td className="px-10 py-8 align-top text-left">
                          <select value={item.subCategoryId} onChange={(e) => updateLineItem(item.id, 'subCategoryId', e.target.value)} className="w-full bg-transparent border-none text-xs font-black text-indigo-700 p-0 focus:ring-0 cursor-pointer"><option value="">Standard Rate</option>{config.categories[item.categoryId]?.subcategories && Object.values(config.categories[item.categoryId].subcategories).map(sub => <option key={sub.id} value={sub.id}>{sub.label}</option>)}</select>
                        </td>
                        <td className="px-10 py-8 align-top text-left">
                          <textarea rows="2" value={item.description} onChange={(e) => updateLineItem(item.id, 'description', e.target.value)} className="w-full bg-transparent border-none text-xs font-extrabold text-slate-500 p-0 focus:ring-0 resize-none outline-none" placeholder="Specs..." />
                        </td>
                        <td className="px-10 py-8 align-top text-center text-left">
                          <input type="number" value={item.quantity} onChange={(e) => updateLineItem(item.id, 'quantity', e.target.value)} className="w-full bg-transparent border-none text-2xl font-black text-center p-0 text-slate-900 focus:ring-0" />
                        </td>
                        <td className="px-6 py-8 align-top text-right text-left">
                          <button onClick={() => setLineItems(lineItems.filter(l => l.id !== item.id))} className="text-slate-200 hover:text-red-500 transition-colors p-2"><Trash2 size={24} /></button>
                        </td>
                      </tr>
                    ))}
                  </tbody>
                </table>
              </div>
            </div>

            {/* Site Constraints Panel */}
            <div className="bg-white rounded-3xl shadow-sm border border-slate-200 p-10 relative text-left">
               <h2 className="font-black flex items-center gap-3 mb-10 text-slate-900 uppercase tracking-widest text-xs text-left">
                <HardHat size={22} className="text-indigo-600" /> Operational Constraints
              </h2>
              <div className="grid grid-cols-1 lg:grid-cols-2 gap-12 text-left">
                <div className="space-y-8 text-left">
                  <div className="flex flex-col gap-5 p-10 bg-slate-50 rounded-[3rem] border border-slate-100 hover:border-indigo-200 transition-all text-left">
                    <div className="flex items-center justify-between text-left">
                      <span className="text-sm font-black text-slate-900 uppercase tracking-widest flex items-center gap-4 text-left">
                        <Calendar size={22} className="text-indigo-600" /> Planned Install Days
                      </span>
                      <div className="flex items-center gap-5 text-left">
                        <input 
                          type="number" 
                          min="0.5" 
                          step="0.5"
                          value={variables.duration}
                          onChange={(e) => setVariables({...variables, duration: parseFloat(e.target.value) || 1})}
                          className="w-28 bg-white border-2 border-indigo-100 rounded-[1.5rem] py-3 px-4 text-center text-xl font-black text-indigo-700 outline-none focus:border-indigo-600"
                        />
                        <span className="text-xs font-black text-slate-400 uppercase tracking-widest">Days</span>
                      </div>
                    </div>
                  </div>

                  <div className="grid grid-cols-1 sm:grid-cols-2 gap-6 text-left">
                    {[
                      { id: 'isUnion', label: 'Union Force' },
                      { id: 'noElevator', label: 'Stair Carry' }
                    ].map(v => (
                      <label key={v.id} className="flex items-center justify-between p-8 border-2 border-slate-50 rounded-3xl hover:bg-white hover:border-indigo-100 transition-all cursor-pointer group shadow-sm text-left">
                        <span className="text-xs font-black text-slate-700 uppercase tracking-widest group-hover:text-indigo-600 transition-colors text-left">{v.label}</span>
                        <input type="checkbox" checked={variables[v.id]} onChange={(e) => setVariables({...variables, [v.id]: e.target.checked})} className="w-8 h-8 rounded-xl text-indigo-600 border-slate-300 focus:ring-indigo-600 cursor-pointer" />
                      </label>
                    ))}
                  </div>
                </div>

                <div className="space-y-8 text-left">
                  <div className="grid grid-cols-1 sm:grid-cols-2 gap-6 text-left">
                    {[
                      { id: 'afterHours', label: 'After Hours' },
                      { id: 'siteProtection', label: 'Protection' }
                    ].map(v => (
                      <label key={v.id} className="flex items-center justify-between p-8 border-2 border-slate-50 rounded-3xl hover:bg-white hover:border-indigo-100 transition-all cursor-pointer group shadow-sm text-left">
                        <span className="text-xs font-black text-slate-700 uppercase tracking-widest group-hover:text-indigo-600 transition-colors text-left">{v.label}</span>
                        <input type="checkbox" checked={variables[v.id]} onChange={(e) => setVariables({...variables, [v.id]: e.target.checked})} className="w-8 h-8 rounded-xl text-indigo-600 border-slate-300 focus:ring-indigo-600 cursor-pointer" />
                      </label>
                    ))}
                  </div>
                  <div className="bg-indigo-950 border border-indigo-900 rounded-[3rem] p-10 flex items-center gap-8 shadow-xl text-left">
                     <div className="bg-indigo-800 p-6 rounded-3xl shadow-inner text-left"><HardHat className="text-indigo-100" size={44} /></div>
                     <div className="space-y-2 text-left">
                        <p className="text-xs font-black text-indigo-300 uppercase tracking-[0.3em] text-left">Target Crew Capacity</p>
                        <p className="text-4xl text-white font-black leading-none text-left">{estimate.crewSize} <span className="text-lg font-bold text-indigo-400 uppercase tracking-widest ml-2">Installers</span></p>
                        <p className="text-[11px] text-indigo-500 font-bold uppercase tracking-wider text-left">Daily shift allocation</p>
                     </div>
                  </div>
                </div>
              </div>
            </div>
          </main>
        )}

        {/* Project Archive Tab */}
        {activeView === 'PROJECTS' && (
          <main className="max-w-[1400px] mx-auto p-12 space-y-16 pb-32 text-left">
            <header className="flex flex-col md:flex-row md:items-end justify-between gap-10 text-left border-b border-slate-200 pb-12">
              <div>
                <h2 className="text-6xl font-black text-slate-900 tracking-tighter uppercase italic leading-none text-left">Project History</h2>
                <p className="text-lg font-bold text-slate-500 mt-4 tracking-tight">Post-job labor historicals and performance variance.</p>
              </div>
              <div className="relative w-full md:w-[450px] shadow-2xl rounded-3xl overflow-hidden text-left">
                <Search className="absolute left-6 top-5 text-slate-300 text-left" size={24} />
                <input type="text" placeholder="Search archive..." value={searchTerm} onChange={(e) => setSearchTerm(e.target.value)} className="w-full pl-16 pr-8 py-5 bg-white border-none rounded-3xl outline-none text-lg font-black text-slate-900 text-left" />
              </div>
            </header>
            <div className="grid grid-cols-1 gap-10 text-left">
              {savedEstimates.filter(est => est.customer.name.toLowerCase().includes(searchTerm.toLowerCase())).sort((a, b) => b.timestamp - a.timestamp).map((est) => {
                  const estLabor = parseFloat(est.estimate.totalCost);
                  const actLabor = parseFloat(est.actualCost) || 0;
                  const variance = estLabor - actLabor;
                  const varianceColor = variance >= 0 ? 'text-emerald-600' : 'text-red-600';
                  return (
                    <div key={est.id} className="bg-white rounded-[4rem] border border-slate-200 p-12 shadow-sm group text-left">
                      <div className="flex flex-col lg:flex-row lg:items-center justify-between gap-12 mb-12 text-left">
                        <div className="flex items-center gap-10 flex-1 w-full text-left">
                          <div className="w-24 h-24 bg-indigo-50 rounded-[2rem] flex items-center justify-center text-indigo-600 shadow-inner group-hover:scale-110 transition-all duration-500 text-left"><FileText size={40} /></div>
                          <div className="text-left">
                            <div className="flex flex-wrap items-center gap-6 mb-4 text-left">
                              <h3 className="text-3xl font-black text-slate-900 text-left">{est.customer.name}</h3>
                              <select value={est.status} onChange={(e) => updateStatus(est.id, e.target.value)} className={`px-6 py-2 rounded-2xl text-[11px] font-black uppercase tracking-widest border-none outline-none shadow-md cursor-pointer text-left ${STATUS_OPTIONS.find(s => s.id === est.status)?.color}`}>{STATUS_OPTIONS.map(opt => <option key={opt.id} value={opt.id}>{opt.label}</option>)}</select>
                            </div>
                            <p className="text-sm text-slate-500 font-black uppercase flex flex-wrap items-center gap-8 tracking-widest text-left text-left">
                              <span className="flex items-center gap-3 text-left"><MapPin size={18} className="text-indigo-600" /> {est.customer.zipCode || 'No Zip'}</span>
                              <span className="flex items-center gap-3 text-left"><Clock size={18} className="text-indigo-600" /> {new Date(est.timestamp).toLocaleDateString()}</span>
                            </p>
                          </div>
                        </div>
                        <div className="flex items-center gap-12 bg-slate-50 p-8 rounded-[2.5rem] text-left">
                           <div className="text-right text-left">
                              <p className="text-xs font-black text-slate-400 uppercase mb-2 tracking-widest text-left">Proposal Price</p>
                              <p className="text-3xl font-black text-indigo-600 tracking-tight text-left">${Number(est.estimate.salesPrice).toLocaleString()}</p>
                           </div>
                           <div className="flex gap-4 text-left">
                              <button onClick={() => loadEstimateFromHistory(est)} className="p-5 bg-white text-indigo-700 rounded-[1.5rem] hover:bg-indigo-600 hover:text-white transition-all shadow-sm border border-indigo-50 text-left"><ExternalLink size={28} /></button>
                              <button onClick={() => deleteEstimate(est.id)} className="p-5 bg-white text-slate-300 hover:text-red-600 rounded-[1.5rem] transition-all border border-slate-100 shadow-sm text-left"><Trash2 size={28} /></button>
                           </div>
                        </div>
                      </div>
                      <div className="bg-slate-50 rounded-[3rem] p-10 grid grid-cols-1 md:grid-cols-3 gap-12 border border-slate-100 text-left">
                        <div className="space-y-4 text-left text-left"><label className="text-xs font-black text-slate-400 uppercase tracking-widest text-left">Est. Labor Cost</label><div className="flex items-center gap-4 text-slate-800 font-black text-2xl justify-start text-left"><Calculator size={24} className="opacity-30" /> ${estLabor.toLocaleString()}</div></div>
                        <div className="space-y-4 text-left text-left"><label className="text-xs font-black text-indigo-600 uppercase tracking-widest italic text-left">Actual Final Labor</label><div className="relative text-left"><DollarSign size={20} className="absolute left-5 top-[1.1rem] text-indigo-500" /><input type="number" value={est.actualCost || ''} onChange={(e) => updateActualCost(est.id, e.target.value)} className="w-full pl-12 pr-8 py-5 bg-white border-2 border-indigo-100 rounded-[2rem] text-xl font-black text-indigo-600 outline-none focus:ring-8 focus:ring-indigo-50 transition-all shadow-lg text-left" placeholder="0.00" /></div></div>
                        <div className="space-y-4 text-left text-left"><label className="text-xs font-black text-slate-400 uppercase tracking-widest text-left">Yield Variance</label><div className={`flex items-center gap-4 font-black text-3xl ${varianceColor} justify-start text-left`}>{variance >= 0 ? <TrendingUp size={32} /> : <TrendingDown size={32} />}<span>{variance >= 0 ? '+' : ''}${variance.toLocaleString()}</span></div></div>
                      </div>
                    </div>
                  );
                })}
            </div>
          </main>
        )}

        {/* Admin Config View */}
        {activeView === 'ADMIN' && (
          <main className="max-w-5xl mx-auto p-12 space-y-16 pb-32 text-left text-left">
             <div className="bg-white p-16 rounded-[4rem] border border-slate-200 shadow-sm relative overflow-hidden text-left">
                <h2 className="text-5xl font-black text-slate-900 mb-16 uppercase tracking-tighter italic flex items-center gap-6 text-left"><Settings2 size={48} className="text-indigo-600" /> Administrative Logic</h2>
                <div className="grid grid-cols-1 md:grid-cols-2 gap-12 mb-20 text-left">
                   <div className="p-10 bg-indigo-50 rounded-[3rem] border border-indigo-100 shadow-inner text-left">
                      <h4 className="font-black text-xs uppercase text-indigo-400 mb-10 tracking-[0.3em] flex items-center gap-4 text-left"><Percent size={24} /> Default Project OPS Margin</h4>
                      <div className="flex items-center gap-8 text-left">
                        <input type="range" min="0.1" max="0.6" step="0.01" value={config.defaultMargin} onChange={(e) => setConfig(p => ({ ...p, defaultMargin: parseFloat(e.target.value) }))} className="flex-1 h-4 bg-indigo-200 rounded-lg appearance-none cursor-pointer accent-indigo-600" />
                        <div className="w-28 bg-white p-6 rounded-3xl border-2 border-indigo-200 text-center font-black text-2xl text-indigo-700 shadow-xl text-left">{((config.defaultMargin || 0) * 100).toFixed(0)}%</div>
                      </div>
                   </div>
                   <div className="p-10 bg-slate-50 rounded-[3rem] border border-slate-100 shadow-inner text-left">
                      <h4 className="font-black text-xs uppercase text-slate-400 mb-10 tracking-[0.3em] flex items-center gap-4 text-left"><DollarSign size={24} /> Base Labor Hourly Costs</h4>
                      <div className="space-y-6 text-left text-left">
                        {Object.entries(config.rates).map(([key, val]) => (
                          <div key={key} className="flex justify-between items-center p-6 bg-white rounded-3xl border border-slate-100 shadow-sm text-left">
                            <span className="text-xs font-black text-slate-500 uppercase tracking-[0.2em] text-left">{key.replace('_', ' ')}</span>
                            <div className="flex items-center gap-3 font-black text-indigo-600 text-2xl text-right text-left">
                              <span className="text-lg opacity-40 text-left">$</span>
                              <input type="number" value={val} onChange={(e) => setConfig(p => ({ ...p, rates: { ...p.rates, [key]: parseFloat(e.target.value) || 0 } }))} className="w-20 bg-transparent border-none p-0 text-right outline-none text-left" />
                            </div>
                          </div>
                        ))}
                      </div>
                   </div>
                </div>
                <div className="grid grid-cols-1 gap-12 text-left text-left">
                  {Object.values(config.categories).map(cat => (
                    <div key={cat.id} className="p-12 bg-slate-50/50 rounded-[4rem] border border-slate-100 text-left text-left">
                      <div className="flex justify-between items-center mb-12 text-left text-left">
                        <h4 className="font-black text-base uppercase text-slate-900 tracking-[0.4em] text-left">{cat.label}</h4>
                        <div className="h-px bg-slate-200 flex-1 mx-10 opacity-40 text-left text-left"></div>
                        <button onClick={() => setConfig(p => ({ ...p, categories: { ...p.categories, [cat.id]: { ...cat, subcategories: { ...cat.subcategories, [`SUB_${Date.now()}`]: { id: `SUB_${Date.now()}`, label: 'New Product', rate: 1.0 } } } } }))} className="bg-white p-4 rounded-2xl shadow-md text-indigo-600 border border-indigo-100 hover:scale-110 transition-transform text-left"><Plus size={24} /></button>
                      </div>
                      <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-4 gap-8 text-left text-left text-left">
                        {Object.values(cat.subcategories || {}).map(sub => (
                          <div key={sub.id} className="bg-white p-8 rounded-[2.5rem] border-2 border-slate-50 shadow-sm hover:border-indigo-400 transition-all group text-left text-left">
                            <div className="flex justify-between items-start mb-6 text-left text-left text-left">
                              <p className="text-[11px] font-black text-slate-400 uppercase tracking-widest text-left text-left">{sub.label}</p>
                              <button onClick={() => {
                                const newSubs = { ...cat.subcategories };
                                delete newSubs[sub.id];
                                setConfig(p => ({ ...p, categories: { ...p.categories, [cat.id]: { ...cat, subcategories: newSubs } } }));
                              }} className="opacity-0 group-hover:opacity-100 text-slate-300 hover:text-red-500 transition-all text-left text-left"><X size={16} /></button>
                            </div>
                            <div className="flex items-center gap-3 font-black text-indigo-600 text-3xl text-left text-left text-left">
                              <input type="number" step="0.1" value={sub.rate} onChange={(e) => setConfig(p => ({ ...p, categories: { ...p.categories, [cat.id]: { ...p.categories[cat.id], subcategories: { ...p.categories[cat.id].subcategories, [sub.id]: { ...sub, rate: parseFloat(e.target.value) || 0 } } } } }))} className="w-full bg-transparent outline-none focus:ring-0 text-left text-left" />
                              <span className="text-sm text-slate-400 font-bold uppercase tracking-widest text-left text-left">H</span>
                            </div>
                          </div>
                        ))}
                      </div>
                    </div>
                  ))}
                </div>
             </div>
          </main>
        )}
      </div>

      {/* Vision Engine Modal */}
      {isAiModalOpen && (
        <div className="fixed inset-0 bg-slate-950/95 backdrop-blur-3xl z-[200] flex items-center justify-center p-8 text-left text-left text-left text-left text-left">
          <div className="bg-white rounded-[5rem] shadow-2xl max-w-3xl w-full overflow-hidden border border-slate-100 text-left text-left text-left text-left">
            <div className="p-12 border-b border-slate-50 flex justify-between items-center bg-indigo-50/20 text-left text-left text-left">
              <div className="flex items-center gap-6 text-left text-left">
                <div className="bg-indigo-600 p-5 rounded-[2rem] text-white shadow-2xl rotate-3 shadow-indigo-300 text-left text-left"><Sparkles size={40} /></div>
                <div className="text-left text-left text-left text-left">
                  <h3 className="text-4xl font-black text-slate-900 tracking-tighter leading-none mb-2 text-left text-left">Aries Vision Engine</h3>
                  <p className="text-xs font-black text-slate-400 uppercase tracking-[0.3em] text-left text-left text-left">{lastFile ? "Refinement Mode" : "Universal Spec Scan"}</p>
                </div>
              </div>
              <button onClick={() => !isAnalyzing && setIsAiModalOpen(false)} className="p-6 text-slate-300 hover:text-slate-900 transition-all duration-300 text-left text-left"><X size={48} /></button>
            </div>
            <div className="p-20 text-left text-left text-left">
              {!isAnalyzing ? (
                <div className="space-y-16 text-left text-left text-left">
                  {lastFile ? (
                    <div className="space-y-12 text-left text-left text-left text-left text-left">
                      <div className="flex items-center gap-6 bg-indigo-50 p-8 rounded-[2.5rem] border-2 border-indigo-100 text-left text-left text-left text-left">
                        <div className="bg-white p-4 rounded-[1.5rem] shadow-md text-indigo-600 text-left text-left"><File size={32} /></div>
                        <div className="text-left text-left text-left text-left text-left">
                          <p className="text-xl font-black text-slate-900 truncate max-w-[400px] text-left text-left"> {lastFile.name}</p>
                          <p className="text-xs text-slate-400 uppercase font-black tracking-widest mt-1 text-left text-left">Cached for refinement</p>
                        </div>
                      </div>
                      <div className="space-y-6 text-left text-left text-left text-left text-left">
                        <label className="text-xs font-black text-slate-500 uppercase tracking-[0.3em] flex items-center gap-4 px-4 text-left text-left">
                          <MessageSquare size={20} className="text-indigo-600 text-left text-left" /> Refinement Instructions
                        </label>
                        <textarea placeholder="e.g. 'Fix workstations to 15'..." value={correctionComments} onChange={(e) => setCorrectionComments(e.target.value)} className="w-full h-56 p-8 bg-slate-50 border-4 border-slate-100 rounded-[3rem] text-lg font-bold outline-none focus:border-indigo-600 transition-all text-left text-left" />
                      </div>
                      <button onClick={handleAiRefine} disabled={!correctionComments} className="w-full bg-indigo-600 hover:bg-slate-950 text-white py-8 rounded-[2.5rem] font-black uppercase tracking-[0.5em] text-sm shadow-2xl active:scale-95 disabled:opacity-30 transition-all text-left shadow-indigo-900/40 text-left">Submit Refinement</button>
                    </div>
                  ) : (
                    <div className="space-y-16 text-center text-left text-left">
                      <div className="mx-auto w-40 h-40 bg-indigo-50 rounded-[4rem] flex items-center justify-center shadow-inner group text-left text-left"><Upload size={80} className="text-indigo-600 transition-transform group-hover:-translate-y-6 text-left text-left" /></div>
                      <div className="space-y-6 text-left text-left text-left">
                        <h4 className="font-black text-slate-900 text-5xl mb-2 tracking-tighter italic text-left text-left">Deep Spec Scan</h4>
                        <p className="text-lg text-slate-500 font-bold leading-relaxed px-20 text-left">Automated interpretation of Project PDFs and manufacturer quotes.</p>
                      </div>
                      <label className="w-full inline-block cursor-pointer bg-slate-950 text-white px-16 py-8 rounded-[3rem] font-black hover:bg-indigo-700 transition-all uppercase tracking-[0.5em] text-sm shadow-2xl active:scale-95 text-center text-left">Select Spec Document<input type="file" className="hidden" accept="image/*,application/pdf" onChange={handleAiUpload} /></label>
                    </div>
                  )}
                </div>
              ) : (
                <div className="text-center py-20 text-left">
                  <div className="relative mx-auto w-56 h-56 mb-16 text-left">
                    <div className="absolute inset-0 border-[20px] border-indigo-50 rounded-full text-left"></div>
                    <div className="absolute inset-0 border-[20px] border-indigo-600 border-t-transparent rounded-full animate-spin text-left"></div>
                    <div className="absolute inset-0 flex items-center justify-center text-indigo-600 text-left">
                       <Sparkles className="animate-pulse text-left" size={80} />
                    </div>
                  </div>
                  <h4 className="font-black text-slate-900 text-4xl mb-6 tracking-tight italic uppercase tracking-widest text-center text-left">Optimizing Dataset...</h4>
                </div>
              )}
            </div>
          </div>
        </div>
      )}
    </div>
  );
}
